204 bool ConsoleHandler::selectHelper()
205 {
206  std::string line;
207  std::getline(std::cin, line);
208 
209  //check if the input is valid
210  std::string assertOrder = "(order\\s+by\\s+\\w+\\s*(asc|desc)?)?$";
211  std::string assertWhere = "where(\\s+\\w+\\s*[!=<>]{1,2}\\s*[\\w/]+\\s+(and|or|" + assertOrder + "))+";
212  std::string assertFrom = "from\\s+\\w+\\s+(" + assertWhere + ")";
213 
214  if (!assert_input(line, "(^\\s*(distinct)?\\s*(\\*|\\s*\\w+\\s*(,|" + assertFrom + "))[^$])"))
215  return false;
216 
217  //break down the input
218  std::string distinct = findMatch(line, "^\\s*(distinct)");
219  std::string table = findMatch(line, "\\s*from\\s+(\\w+)");
220  std::string query = findMatch(line, "\\s*from\\s+" + table + "\\s+where([\\w\\s!=<>/]+)$");
221  std::string order = findMatch(line, "\\s*from\\s+" + table + ".*?(order\\s+by[\\s\\w]*)$");
222  std::string byWhat = findMatch(order, "\\s*order\\s+by\\s+(\\w+)\\s*(asc|desc)?$");
223  std::string cend = findMatch(order, "\\s*order\\s+by\\s+" + byWhat + "\\s+(asc|desc)$");
224 
225 
226  bool isDistinct = (!distinct.empty());
227  bool asc = (cend == "asc");
228 
229  std::string select;
230  if (isDistinct) select = findMatch(line, "^\\s*distinct([\*\\w\\s,]*)from\\s*" + table +
231  "(\\s+where\\s*" + query + ")?\\s*$");
232  else select = findMatch(line, "^\\s*([\*\\w\\s,]*)from\\s*" + table +
233  "(\\s+where\\s*" + query + ")?\\s*");
234 
235  if (!order.empty()) query = findMatch(query, "\\s*(.*?)" + order + "$");
236 
237  //get selected columns, if * it will be empty
238  table_row cols = findMatches(select, "\\w+");
239 
240  //assert that if there is a column to sort by
241  //it is selected
242  if (!cols.empty() && !byWhat.empty()) {
243  bool found = false;
244  for (const std::string& col : cols)
245  if (col == byWhat) found = true;
246 
247  if (!found) {
248  std::cout << "Cannot order by column which is not selected! ";
249  return false;
250  }
251  }
252 
253  db->select(table, query, cols, isDistinct, byWhat, asc);
254 }
